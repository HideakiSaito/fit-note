def create
  @equipment = Equipment.new(equipment_params)
  if @equipment.save
    status = 'success'
    html = render_to_string partial: 'show',
      locals: { equipment: @equipment }
  else
    status = 'error'
  end
  render json: { status: status, data: @equipment, html:html }
